Publish to GitHub LIVE WEBSITE

Your live site at nextvisiontulsa.com updates whenever you upload a fresh TulsaFilmTribe.html to your GitHub repository. Here's the workflow:

1
Make your changes here
Add members, approve edits, update copy — all in this admin panel. Changes save instantly to this device only.
2
Click "Publish Tribe" (top of page)
Downloads a single TulsaFilmTribe.html file with all your data baked in. Save it to your Desktop.
3
Upload the file to GitHub
Open your repo, drag the new TulsaFilmTribe.html on top of the old one, click "Commit changes". Done.
4
Wait ~60 seconds, then refresh the live site
GitHub Pages rebuilds automatically. Hard-refresh (Cmd+Shift+R / Ctrl+Shift+R) to bust the cache.
New to GitHub? Click here to learn ↓

GitHub is a free service that hosts your website's files. It's like Dropbox, but built for code — and with a feature called GitHub Pages that turns any HTML file into a real website with a real URL.

How your setup works:

  • You have a repository (a folder) on GitHub that holds TulsaFilmTribe.html
  • GitHub Pages serves that file at nextvisiontulsa.com
  • Whenever you upload a new version, the live site updates within ~60 seconds

First-time GitHub setup (only do this once):

  1. Go to github.com/signup and create a free account
  2. Create a new repository (any name — e.g. tribe-site) and make it public
  3. Click "uploading an existing file" and drop in your downloaded TulsaFilmTribe.html
  4. Rename it to index.html so it loads as the homepage
  5. Settings → Pages → Source: main branch → Save
  6. Settings → Pages → add your custom domain nextvisiontulsa.com

After that, every update is just: Publish → drag file onto repo → Commit. That's it.

⚠ Important: The downloaded file is named TulsaFilmTribe.html. On GitHub, your live file is probably called index.html. Rename when uploading — drag-replace works if names match.

-- Tulsa Film Tribe — Supabase schema
-- Run this in Supabase SQL Editor once per project.

create table if not exists tribe_members (
  id text primary key,
  name text not null,
  role text,
  category int default 0,
  bio text,
  skills text,
  instagram text,
  email text,
  photo_url text,
  sort_order int default 0,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

create table if not exists tribe_codes (
  code text primary key,
  label text,
  member_id text references tribe_members(id) on delete set null,
  uses int,
  uses_count int default 0,
  revoked boolean default false,
  created_at timestamptz default now(),
  last_used timestamptz
);

create table if not exists tribe_pending_edits (
  id text primary key,
  member_id text references tribe_members(id) on delete cascade,
  code text,
  changes jsonb,
  status text default 'pending',
  submitted_at timestamptz default now(),
  decided_at timestamptz
);

create table if not exists tribe_apps (
  id text primary key,
  name text,
  email text,
  discipline text,
  about text,
  why text,
  status text default 'new',
  submitted_at timestamptz default now()
);

create table if not exists tribe_content (
  key text primary key,
  value jsonb,
  updated_at timestamptz default now()
);

-- Storage bucket for photos
insert into storage.buckets (id, name, public)
values ('tribe-photos', 'tribe-photos', true)
on conflict (id) do nothing;

-- Row Level Security — public read, authenticated write.
-- For v1 we allow anon write too so admin works without auth.
-- TIGHTEN THIS LATER by adding Supabase Auth to the admin.

alter table tribe_members enable row level security;
alter table tribe_codes enable row level security;
alter table tribe_pending_edits enable row level security;
alter table tribe_apps enable row level security;
alter table tribe_content enable row level security;

create policy "public read members"       on tribe_members       for select using (true);
create policy "public write members"      on tribe_members       for all    using (true) with check (true);
create policy "public read codes"         on tribe_codes         for select using (true);
create policy "public write codes"        on tribe_codes         for all    using (true) with check (true);
create policy "public read pending"       on tribe_pending_edits for select using (true);
create policy "public write pending"      on tribe_pending_edits for all    using (true) with check (true);
create policy "public read apps"          on tribe_apps          for select using (true);
create policy "public write apps"         on tribe_apps          for all    using (true) with check (true);
create policy "public read content"       on tribe_content       for select using (true);
create policy "public write content"      on tribe_content       for all    using (true) with check (true);

create policy "public read tribe-photos"  on storage.objects     for select using (bucket_id = 'tribe-photos');
create policy "public write tribe-photos" on storage.objects     for all    using (bucket_id = 'tribe-photos') with check (bucket_id = 'tribe-photos');
